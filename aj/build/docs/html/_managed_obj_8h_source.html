122  } else {
123  /* Normal copy constructor (inc ref) of existing object */

440  void DecRef()
441  {
442  uint32_t refs = DecrementAndFetch(&context->refCount);
443  if (0 == refs) {
444  /* Call the overriden destructor */
445  object->~T();

447  free(context);
