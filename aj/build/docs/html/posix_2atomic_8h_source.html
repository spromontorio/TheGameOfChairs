47 inline int32_t IncrementAndFetch(volatile int32_t* mem)

49  /*
50  * Androids built in __atomic_inc operation returns the previous value
51  * stored in mem. This value has been changed in memory. Since we expect
52  * the value after the increment to be returned we add one to the value
